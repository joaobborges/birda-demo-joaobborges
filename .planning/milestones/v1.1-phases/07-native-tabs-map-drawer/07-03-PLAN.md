**2. Wire BottomSheet into `app/(main)/index.tsx`:**

After Plan 01, the map screen has `selectedBird` state but no BirdInfoCard. Now add the BottomSheet:

Add imports:
```tsx
import { useRef, useCallback } from 'react'
import BottomSheet, { BottomSheetView } from '@gorhom/bottom-sheet'
import { useRouter } from 'expo-router'
import { BirdDrawerContent } from '@/components/map/BirdDrawerContent'
import { colors } from '@/theme/colors'
```

Add ref and callbacks:
```tsx
const sheetRef = useRef<BottomSheet>(null)
const router = useRouter()

// Called when a bird marker is tapped
function handleBirdPress(bird: Bird) {
  setSelectedBird(bird)
  sheetRef.current?.snapToIndex(0)  // open to 50%
}

// Called when sheet is dismissed (swipe down)
const handleSheetClose = useCallback(() => {
  setSelectedBird(null)
}, [])

// Called when bird image is tapped in drawer
function handleBirdImagePress() {
  if (!selectedBird) return
  sheetRef.current?.close()
  router.push({ pathname: '/bird-detail', params: { birdId: selectedBird.id } })
}
```

Update `BirdMarker` onPress to use `handleBirdPress` instead of `setSelectedBird`.

Add BottomSheet as the LAST child in the container View (after MapView and floating top bar), so it renders above everything:
```tsx
<BottomSheet
  ref={sheetRef}
  index={-1}
  snapPoints={['50%']}
  enablePanDownToClose
  backdropComponent={null}
  onClose={handleSheetClose}
  style={{ marginHorizontal: 0 }}
  handleIndicatorStyle={{ backgroundColor: colors.neutral300 }}
>
  <BottomSheetView style={{ flex: 1 }}>
    {selectedBird && (
      <BirdDrawerContent
        bird={selectedBird}
        onImagePress={handleBirdImagePress}
      />
    )}
  </BottomSheetView>
</BottomSheet>
```

Key configuration per research:
- `index={-1}` — starts closed (NAV-05)
- `snapPoints={['50%']}` — single snap point at half screen
- `enablePanDownToClose` — swipe-to-dismiss (NAV-05)
- `backdropComponent={null}` — no overlay, map stays interactive
- `style={{ marginHorizontal: 0 }}` — full width, touches left/right edges (NAV-03)
- BottomSheet is last child so it renders above floating icons (NAV-04)

Do NOT conditionally mount/unmount the BottomSheet component — keep it always mounted, control visibility via snapToIndex/close.

Remove the `BirdInfoCard` import if still present after Plan 01.
